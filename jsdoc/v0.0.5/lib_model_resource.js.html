'use strict';

const Identifiable = require('./identifiable');
const ModelUtil = require('../modelutil');

const Field = require('../parser/field');

const Relationship = require('./relationship');
const RelationshipDeclaration = require('../parser/relationshipdeclaration');

const ClassDeclaration = require('../parser/classdeclaration');

const EnumValueDeclaration = require('../parser/enumvaluedeclaration');
const EnumDeclaration = require('../parser/enumdeclaration');

/**
 * Resource is an instance that is associated with a ClassDeclaration and
 * that has a set of Properties that are defined by the ClassDeclaration.
 *
 * It is used in Concerto to represent Assets, Participants, Transactions and
 * other domain classes that can be serialized for long-term persistent storage.
 * @extends Identifiable
 */
class Resource extends Identifiable {
    /**
     * This constructor should not be called directly.
     * Use the Factory class to create instances.
     *
     * @param {ModelManager} modelManager - The ModelManager for this instance
     * @param {string} ns - The namespace this instance.
     * @param {string} type - The type this instance.
     * @param {string} id - The identifier of this instance.
     * @private
     */
    constructor(modelManager, ns, type, id) {
        super(modelManager, ns, type, id);
    }

    /**
     * Sets a property, validating that it does not violate the model
     * @param {string} propName - the name of the field
     * @param {string} value - the value of the property
     * @throws {Error} if the value is not compatible with the model definition for the field
     */
    setPropertyValue(propName, value) {
        let classDeclaration = this.getClassDeclaration();
        let field = classDeclaration.getProperty(propName);

        if (!field) {
            throw new Error('The instance with id ' +
                this.getIdentifier() + ' trying to set field ' +
                propName + ' which is not declared in the model.');
        }

        this.validateProperty(this.getIdentifier(), field, value);
        this[propName] = value;
    }

    /**
     * Sets a property, validating that it does not violate the model
     * @param {string} propName - the name of the field
     * @param {string} value - the value of the property
     * @throws {Error} if the value is not compatible with the model definition for the field
     */
    addArrayValue(propName, value) {
        let classDeclaration = this.getClassDeclaration();
        let field = classDeclaration.getProperty(propName);

        if (!field) {
            throw new Error('The instance with id ' +
                this.getIdentifier() + ' trying to set field ' +
                propName + ' which is not declared in the model.');
        }

        if (!field.isArray()) {
            throw new Error('The instance with id ' +
                this.getIdentifier() + ' trying to add array item ' +
                propName + ' which is not declared as an array in the model.');
        }

        this.validateNonArrayProperty(this.getIdentifier(), field, value);
        if(this[propName]) {
            this[propName].push(value);
        }
        else {
            this[propName] = [value];
        }
    }

    /**
     * Sets the fields to their default values, based on the model
     * @private
     */
    assignFieldDefaults() {
        let classDeclaration = this.getClassDeclaration();
        let fields = classDeclaration.getProperties();

        for (let n = 0; n < fields.length; n++) {
            let field = fields[n];
            if (field instanceof Field) {
                let defaultValue = field.getDefaultValue();

                if (defaultValue) {
                    if (field.getType() === 'String') {
                        this.setPropertyValue(field.getName(), defaultValue);
                    } else if (field.getType() === 'Integer') {
                        this.setPropertyValue(field.getName(), parseInt(defaultValue));
                    } else if (field.getType() === 'Long') {
                        this.setPropertyValue(field.getName(), parseInt(defaultValue));
                    } else if (field.getType() === 'Double') {
                        this.setPropertyValue(field.getName(), parseFloat(defaultValue));
                    } else if (field.getType() === 'Boolean') {
                        this.setPropertyValue(field.getName(), (defaultValue === 'true'));
                    } else if (field.getType() === 'DateTime') {
                        const dateTime = new Date();
                        dateTime.setTime(Date.parse(defaultValue));
                        this.setPropertyValue(field.getName(), dateTime);
                    }
                }
            }
        }
    }

    /**
     * Returns the class declaration for this instance object.
     *
     * @return {ClassDeclaration} - the class declaration for this Resource
     * @throws {Error} - if the class or namespace for the instance is not declared
     * @private
     */
    getClassDeclaration() {
        // do we have a model file?
        let modelFile = this.modelManager.getModelFile(this.getNamespace());

        if (!modelFile) {
            throw new Error('No model for namespace ' + this.getNamespace() + ' is registered with the ModelManager');
        }

        // do we have a class?
        let classDeclaration = modelFile.getType(this.getType());

        if (!classDeclaration) {
            throw new Error('The namespace ' + this.getNamespace() + ' does not contain the type ' + this.getType());
        }

        return classDeclaration;
    }

    /**
     * Validates the instance against its model.
     *
     * @throws {Error} - if the instance if invalid with respect to the model
     * @private
     */
    validate() {
        let classDeclaration = this.getClassDeclaration();

        // check all the properties are declared
        let props = Object.getOwnPropertyNames(this);

        for (let n = 0; n < props.length; n++) {
            let propName = props[n];
            // we don't validate system properties
            if (!Identifiable.isSystemProperty(propName)) {
                let field = classDeclaration.getProperty(propName);

                if (!field) {
                    throw new Error('The instance with id ' +
                        this.getIdentifier() + ' has a field named ' +
                        propName + ' which is not declared in the model.');
                }

                // validate the data agains the field from the model
                let data = this[propName];
                this.validateProperty(this.getIdentifier(), field, data);
            }
        }

        // are there any missing non-optional fields?
        let fields = classDeclaration.getProperties();

        for (let n = 0; n < fields.length; n++) {
            let field = fields[n];
            if (field instanceof Field && field.isOptional() === false) {
                if (this.hasOwnProperty(field.getName()) === false || this[field.getName()] === null) {
                    Resource.reportMissingRequiredProperty(this.getIdentifier(), field);
                }
            }
        }
    }

    /**
     * Validates a field / data combination against the model
     *
     * @param {string} id - the identifier of this instance.
     * @param {Field} field - the field
     * @param {*} data - the data to validate against the field
     * @throws {Error} - if the data is invalid with respect to the field
     * @private
     */
    validateProperty(id, field, data) {
        let dataType = typeof data;
        let propName = field.getName();

        if (dataType === 'undefined' || dataType === 'symbol' || dataType === 'function') {
            Resource.reportModelViolation(id, propName, data, field);
        }

        if (data !== null) {
            if (field.isArray() === false) {
                this.validateNonArrayProperty(id,field,data);
            } else {
                // an array type
                if (data.constructor !== Array) {
                    Resource.reportModelViolation(id, propName, data, field);
                }
                else {
                    // should have a length property
                    if(!data.length) {
                        Resource.reportModelViolation(id, propName, data, field);
                    }
                    else {
                        // check the contents of the array
                        for(let n=0; n < data.length; n++) {
                            const dataItem = data[n];
                            this.validateNonArrayProperty(id,field,dataItem);
                        }
                    }
                }
            }
        } // data !== null
    }

    /**
     * Validates a non-array field / data combination against the model
     *
     * @param {string} id - the identifier of this instance.
     * @param {Property} property - the property
     * @param {*} data - the data to validate against the field
     * @throws {Error} - if the data is invalid with respect to the field
     * @private
     */
    validateNonArrayProperty(id, property, data) {
        const dataType = typeof data;
        const propName = property.getName();

        // console.log( 'dataType is ' + dataType );
        // console.log( 'property is ' + property.getName() );
        // const isField = property instanceof Field;
        // console.log( '** property instanceof Field: ' + isField );

        if (property instanceof Field) {
            switch (property.getType()) {
            case 'Integer':
            case 'Double':
            case 'Long':
                if (dataType !== 'number') {
                    Resource.reportModelViolation(id, propName, data, property);
                }
                break;

            case 'Boolean':
                if (dataType !== 'boolean') {
                    Resource.reportModelViolation(id, propName, data, property);
                }
                break;

            case 'DateTime':
                if (data instanceof Date === false) {
                    Resource.reportModelViolation(id, propName, data, property);
                }
                break;

            case 'String':
                if (dataType !== 'string') {
                    Resource.reportModelViolation(id, propName, data, property);
                }
                break;
            default:
                // e.g. this could be a Field that references an Enum
                // or a reference to another complex type (asset/transaction)
                {
                    // see if the type we reference is an enum
                    const myModelFile = this.getModelManager().getModelFile(this.getNamespace());
                    const propertyType = myModelFile.getType(property.getType());
                    if(!propertyType) {
                        Resource.reportModelViolation(id, propName, data, property);
                    }

                    if(propertyType instanceof EnumDeclaration) {
                        if(dataType !== 'string') {
                            // enums must be strings
                            Resource.reportModelViolation(id, propName, data, property);
                        }
                        else {
                            // does the string exist in the enum?
                            if(propertyType.getProperty(data) === null) {
                                Resource.reportModelViolation(id, propName, data, property);
                            }
                        }
                    }
                    else {
                        // check the data conforms to the type
                        // console.log('1 ' + (propertyType instanceof ClassDeclaration));
                        // console.log('2 ' + (data instanceof Identifiable));
                        // console.log('3 ' + ModelUtil.isAssignableTo(myModelFile, data.getFullyQualifiedType(), propertyType));
                        // console.log('propertyType ' + propertyType.getFullyQualifiedName());
                        // console.log('data ' + data.getFullyQualifiedType());

                        if(propertyType instanceof ClassDeclaration &&
                            data instanceof Identifiable &&
                              ModelUtil.isAssignableTo(myModelFile, data.getFullyQualifiedType(), propertyType) && data.validate) {
                            // the data is a complex type, let's validate it!
                            data.validate();
                        }
                        else {
                            Resource.reportModelViolation(id, propName, data, property);
                        }
                    }
                }
            }
        } else if (property instanceof RelationshipDeclaration) {
            if (data instanceof Relationship === false) {
                Resource.reportModelViolation(id, propName, data, property);
                // TODO (DCS) check the type of the relationship against
                // the type of the property
            }
        }
        else if (property instanceof EnumValueDeclaration ) {
            if (data instanceof String === false) {
                Resource.reportModelViolation(id, propName, data, property);
            }
        } else {
            Resource.reportModelViolation(id, propName, data, property);
        }
    }

    /**
     * Throw a new error for a model violation.
     * @param {string} id - the identifier of this instance.
     * @param {string} propName - the name of the field.
     * @param {*} value - the value of the field.
     * @param {Field} field - the field/
     * @private
     */
    static reportModelViolation(id, propName, value, field) {
        let isArray = field.isArray() ? '[]' : '';
        let typeOfValue = typeof value;

        if(value instanceof Identifiable) {
            typeOfValue = value.getFullyQualifiedType();
            value = value.getFullyQualifiedIdentifier();
        }

        throw new Error('Model violation in instance ' + id + ' field ' + propName + ' has value ' + value + ' (' + typeOfValue + ') expected type ' + field.getType() + isArray);
    }

    /**
     * Throw a new error for a missing, but required field.
     * @param {string} id - the identifier of this instance.
     * @param {Field} field - the field/
     * @private
     */
    static reportMissingRequiredProperty(id, field) {
        throw new Error('Instance ' + id + ' missing required field ' + field.getName());
    }
}

module.exports = Resource;
