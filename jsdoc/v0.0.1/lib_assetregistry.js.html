'use strict';

const Registry = require('./registry');
const Util = require('./util');
const uuid = require('node-uuid');

/**
 * The AssetRegistry is used to manage a set of assets.
 * <p><a href="diagrams/assetregistry.svg"><img src="diagrams/assetregistry.svg" style="width:100%;"/></a></p>
 * @extends Registry
 */
class AssetRegistry extends Registry {

    /**
     * Create the AssetRegistry.
     * @param {ModelManager} modelManager - The ModelManager that defines the
     * models (schema) for types in this AssetRegistry
     * @param {Factory} factory - The factory to use for this registry
     * @param {Serializer} serializer - The Serializer to use for this registy
     * @param {string} id - The unique identifier of the registry
     * @param {string} name - The name of the registry
     */
    constructor(modelManager, factory, serializer, id, name) {
        super(modelManager, factory, serializer, id, name);
    }

    /**
     * Get a list of all existing asset registries.
     * @param {SecurityContext} securityContext - The user's security context
     * @param {ModelManager} modelManager - The ModelManager that defines the
     * models (schema) for types in this AssetRegistry
     * @param {Factory} factory - The factory to use for this registry
     * @param {Serializer} serializer - The Serializer to use for this registy
     * @return {Promise} - A promise that will be resolved with a list of existing
     * asset registries
     */
    static getAllAssetRegistries(securityContext, modelManager, factory, serializer) {
        Util.securityCheck(securityContext);
        if (!modelManager) {
            throw new Error('modelManager not specified');
        } else if (!factory) {
            throw new Error('factory not specified');
        } else if (!serializer) {
            throw new Error('serializer not specified');
        }
        return Util.queryChainCode(securityContext, 'getAllAssetRegistries', [])
            .then(function (buffer) {
                let assetRegistries = JSON.parse(buffer.toString());
                return assetRegistries.map(function (assetRegistry) {
                    return new AssetRegistry(modelManager, factory, serializer, assetRegistry.id, assetRegistry.name);
                });
            });
    }

    /**
     * Get an existing asset registry.
     * @param {SecurityContext} securityContext - The user's security context
     * @param {ModelManager} modelManager - The ModelManager that defines the
     * models (schema) for types in this AssetRegistry
     * @param {Factory} factory - The factory to use for this registry
     * @param {Serializer} serializer - The Serializer to use for this registy
     * @param {string} id - The unique identifier of the asset registry
     * @return {Promise} - A promise that will be resolved with the existing asset
     * registry, or rejected if the asset registry does not exist.
     */
    static getAssetRegistry(securityContext, modelManager, factory, serializer, id) {
        Util.securityCheck(securityContext);
        if (!modelManager) {
            throw new Error('modelManager not specified');
        } else if (!factory) {
            throw new Error('factory not specified');
        } else if (!serializer) {
            throw new Error('serializer not specified');
        } else if (!id) {
            throw new Error('id not specified');
        }
        return Util.queryChainCode(securityContext, 'getAssetRegistry', [id])
            .then(function (buffer) {
                let assetRegistry = JSON.parse(buffer.toString());
                return new AssetRegistry(modelManager, factory, serializer, assetRegistry.id, assetRegistry.name);
            });
    }

    /**
     * Add a new asset registry.
     * @param {SecurityContext} securityContext - The user's security context
     * @param {ModelManager} modelManager - The ModelManager that defines the
     * models (schema) for types in this AssetRegistry
     * @param {Factory} factory - The factory to use for this registry
     * @param {Serializer} serializer - The Serializer to use for this registy
     * @param {string} name - The name of the asset registry
     * @return {Promise} - A promise that will be resolved with the new asset
     * registry after it has been added.
     */
    static addAssetRegistry(securityContext, modelManager, factory, serializer, name) {
        Util.securityCheck(securityContext);
        let id = uuid.v4();
        if (!modelManager) {
            throw new Error('modelManager not specified');
        } else if (!factory) {
            throw new Error('factory not specified');
        } else if (!serializer) {
            throw new Error('serializer not specified');
        } else if (!name) {
            throw new Error('name not specified');
        }
        return Util.invokeChainCode(securityContext, 'addAssetRegistry', [id, name])
            .then(function () {
                return new AssetRegistry(modelManager, factory, serializer, id, name);
            });
    }

    /**
     * Adds an element to the registry.
     *
     * @param {SecurityContext} securityContext - The user's security context
     * @param {CObject} asset - the asset to be added to the registry.
     * @throws {SecurityException} if the user context is invalid
     * @return {Promise} a promise that is resolved when the element is added to
     * the registry.
     */
    add(securityContext, asset) {
        super.add(securityContext);
        let json = this.serializer.toJSON(asset);
        return Util.invokeChainCode(securityContext, 'addAssetToRegistry', [this.id, asset.getIdentifier(), json]);
    }

    /**
     * Updates an element in the registry.
     *
     * @param {SecurityContext} securityContext - The user's security context
     * @param {CObject} asset - the asset to be updated in the registry.
     * @throws {SecurityException} if the user context is invalid
     * @return {Promise} a promise that is resolved when the element is updated
     * in the registry.
     */
    update(securityContext, asset) {
        super.update(securityContext);
        let json = this.serializer.toJSON(asset);
        return Util.invokeChainCode(securityContext, 'updateAssetInRegistry', [this.id, asset.getIdentifier(), json]);
    }

    /**
     * Remove an asset with a given type and id from the registry.
     *
     * @param {SecurityContext} securityContext - The user's security context
     * @param {string} id - the asset to be updated in the registry.
     * @throws {SecurityException} if the user context is invalid
     * @return {Promise} a promise that is resolved when the element is removed
     * from the registry.
     */
    remove(securityContext, id) {
        super.remove(securityContext);
        return Util.invokeChainCode(securityContext, 'removeAssetFromRegistry', [this.id, id]);
    }

    /**
     * Get all of the assets in the registry.
     *
     * @param {SecurityContext} securityContext - The user's security context
     * @throws {SecurityException} if the user context is invalid
     * @return {Promise} a promise that is resolved with an array of {@link CObject}
     * instances for each asset in the registry.
     */
    getAll(securityContext) {
        let self = this;
        super.getAll(securityContext);
        return Util.queryChainCode(securityContext, 'getAllAssetsInRegistry', [this.id])
            .then(function (buffer) {
                let assets = JSON.parse(buffer.toString());
                return assets.map(function (asset) {
                    return self.serializer.fromJSON(asset.data);
                });
            });
    }

    /**
     * Get a specific asset in the registry.
     *
     * @param {SecurityContext} securityContext - The user's security context
     * @param {string} id - the identifier of the asset in the registry
     * @throws {SecurityException} if the user context is invalid
     * @return {Promise} a promise that is resolved with a {@link CObject}
     * instance for the asset in the registry.
     */
    get(securityContext, id) {
        let self = this;
        super.getAll(securityContext);
        return Util.queryChainCode(securityContext, 'getAssetInRegistry', [this.id, id])
            .then(function (buffer) {
                let asset = JSON.parse(buffer.toString());
                return self.serializer.fromJSON(asset.data);
            });
    }

}

module.exports = AssetRegistry;
